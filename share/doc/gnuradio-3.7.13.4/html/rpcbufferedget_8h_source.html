39  void offer_data(const TdataType& data) {
40  if (!d_data_needed)
41  return;
42  {
43  boost::mutex::scoped_lock lock(d_buffer_lock);
44  d_buffer = data;
45  d_data_needed = false;
46  }
47  d_data_ready.notify_one();
48  }
49 
50  TdataType get() {

52  d_data_needed = true;
53  d_data_ready.wait(lock);
54  return d_buffer;
55  }

57 private:
