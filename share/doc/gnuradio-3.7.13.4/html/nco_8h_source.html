68  // increment current phase angle
69  void step()
70  {
71  phase += phase_inc;
72  if(fabs(phase) > M_PI) {
73  while(phase > M_PI)
74  phase -= 2*M_PI;
75 
76  while(phase < -M_PI)
77  phase += 2*M_PI;
78  }
79  }
80 
81  void step(int n)
82  {
83  phase += phase_inc * n;
84  if(fabs(phase) > M_PI){
85  while(phase > M_PI)
86  phase -= 2*M_PI;
87 
88  while(phase < -M_PI)
89  phase += 2*M_PI;
90  }
91  }
